% --- Global figure downscale ---
% Every tikzpicture is drawn ~10% smaller: `scale` shrinks the coordinates and
% `transform shape` shrinks the node contents (shapes + text) with them, so nodes,
% arcs, labels and spacing all reduce together, and the bounding box updates (so
% \resizebox figures still divide cleanly). Change the single factor to rescale the
% whole collection; set to 1 to disable. Note: tokens drawn as raw `\fill ... circle
% (Npt)` paths use absolute units, so they shrink slightly less than the nodes.
\tikzset{every picture/.style={scale=0.9, transform shape}}

% --- Callouts: box-free, run-in headings ---
% Same names and {title} call form as the old tcolorbox versions, so the
% section files need no changes. The optional brace group becomes part of
% the heading: "Definition (Title).  body..."

\newcommand{\callout@head}[1]{%
  \@ifnextchar\bgroup
    {\callout@headtitled{#1}}
    {{\bfseries\color{accent}#1.}\hspace{0.4em}\ignorespaces}}
